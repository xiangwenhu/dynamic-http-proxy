GET 请求代码： 
            
                fetch("/proxy/api/weather/city/101030100", {
                    headers:{
                       __proxy__: JSON.stringify({
                             target: "http://t.weather.sojson.com",      
                       })
                    }
              })
              .then( res=> res.json())
              .then(d=>console.log(d))

                fetch("/proxy/aggsite/SubCategories",{
